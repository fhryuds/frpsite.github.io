使用方法#

上传frp和frpc.ini到需要内网穿透的机子(尽量使用和服务器同版本的frp,避免报错），sudo chmod +x frpc(给frpc加上执行权),改写配置文件后运行

./frpc -c ./frpc.ini

使用system控制frp启动#

sudo nano /lib/systemd/system/frpc.service

#新建一个名为frpc的systemctl服务，nano可以换成vim

[Unit] 
Description=frpc service 
After=network.target syslog.target 
Wants=network.target  

[Service] 
Type=simple 
ExecStart=/opt/frpc/frpc -c /opt/frpc/frpc.ini  
#这里把frpc和frpc.ini的路径换成你的实际路径

[Install] 
WantedBy=multi-user.target

保存并退出

启动frpc服务： sudo systemctl start frpc

停止frpc服务： sudo systemctl stop frpc

重启frpc服务： sudo systemctl restart frpc

开机自动运行frpc服务： sudo systemctl enable frpc

配置文件#

编辑frpc.ini文件，将里面的内容清空换成如下

[common]
server_addr = frp-fl.yzhtyu.eu.org
server_port = 7000
token = flameix.top

SSH远程登陆#

[ssh-remote.domain.com]   
# 这里不能与别人重复，建议用域名，或者随便几个数字，字母
type = tcp  
# 使用tcp协议
local_ip = 127.0.0.1   
# 需要穿透的机子运行frp，使用本地地址
local_port = 22   
#ssh默认端口
remote_port = 12345  
# 远程端口（必须是10000-65000）范围中的，且不能与别人重复，随即打几个范围内数字即可
所以ssh连接需要使用： ssh -p 12345(远程端口） root@frp-fl.yzhtyu.eu.org(对应frp服务器）#

windows远程桌面RDP连接#

[win-rdp.domain.com]  
type = udp   
# 使用udp协议
local_ip = 127.0.0.1  
# 需穿透的设备内网IP.
local_port = 3389  
# Windows远程桌面默认内网端口
remote_port = 18932  
# 这里一样不能重复，和上面做了区分

http协议web服务#

[http-web.domain.com]  
# 改下名称，建议搭配域名
type = http   
# 使用http协议
local_ip = 127.0.0.1   
# 需穿透的设备内网IP.
local_port = 80   
# 这里用nginx默认端口，搭建网站端口
custom_domains = web.aaaaa.site   
# 填写你要绑定的域名，而且要添加对应cname记录指向frp-fl.yzhtyu.eu.org

https协议web服务#

[https-web.domain.com]  
# 使用自己的域名，或者随便打几个字母，总之别重复，不然启动不了
type = https  
# 使用https协议
local_ip = 127.0.0.1   
# 需穿透的设备内网IP.
local_port = 443   
# 这里使用nginx端口，但需要申请ssl
custom_domains = https.aaaa.site  
#需要绑定的域名，并添加cname记录
